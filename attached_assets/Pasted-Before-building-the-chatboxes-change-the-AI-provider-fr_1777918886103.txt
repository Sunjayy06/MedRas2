def call_gemini(screen, message, 
                session, api_key):
    import urllib.request, json

Never change the normality decision.

assistant. Help researchers understand their 
analysis plan.
When user asks to ADD a test respond with

{"action": "add_test", "test_id": "[name]", 
"message": "Added [name]."}
When REMOVING:
{"action": "remove_test", "test_id": "[name]",
"message": "Removed [name]."}
Never produce any statistical number.""",
        
        'results': """You are a results 
interpretation assistant. Results are correct 
and cannot be changed.
Never recalculate. Never dispute a result.
If asked to recalculate say: Results were 
produced by validated libraries and cannot 
be changed. I can help you understand them."""
    }
    
    system = system_prompts.get(screen, '')
    context = build_context(screen, session)
    full_prompt = (system + '\n\n' + 
                   context + '\n\n' + 
                   'User: ' + message)
    
    url = ('https://generativelanguage.'
           'googleapis.com/v1beta/models/'
           'gemini-1.5-flash:generateContent'
           '?key=' + api_key)

        'contents': [{
            'parts': [{'text': full_prompt}]
        }],
        'generationConfig': {
            'maxOutputTokens': 400,
            'temperature': 0.3
        }
    }).encode('utf-8')

        url,
        data=payload,
        headers={'Content-Type': 
                 'application/json'},
        method='POST'
    )

        with urllib.request.urlopen(
                req, timeout=10) as resp:

                resp.read().decode())

                    ['content']['parts'][0]

                    try:
                        action = json.loads(
                            match.group())
                        return action
                    except Exception:
                        pass
