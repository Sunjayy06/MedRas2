FUNCTION_MAP = {
    'run_paired_ttest': run_paired_ttest,
    'run_wilcoxon': run_wilcoxon,
    'run_mcnemar': run_mcnemar,
    'run_rm_anova': run_rm_anova,
    'run_friedman': run_friedman,
    'run_chi_or_fisher': run_chi_or_fisher,
    'run_kappa': run_kappa,
    'run_icc_bland_altman': run_icc_bland_altman,
    'run_kaplan_meier': run_kaplan_meier,
    'run_cox_regression': run_cox_regression,
    'run_diagnostic_accuracy': run_diagnostic_accuracy,
    'run_ordinal_logistic': run_ordinal_logistic,
    'run_count_regression': run_count_regression
}

all_results = []

for test in planned_tests:
    fn_name = test['function']
    fn = FUNCTION_MAP.get(fn_name)
    if fn is None:
        continue
    try:
        result = fn(
            session=session,
            df=df,
            **test['args']
        )
        if result:
            result['test_type'] = test['test_type']
            result['plan_name'] = test['name']
            result['plan_reason'] = test['reason']
            all_results.append(result)
    except Exception as e:
        all_results.append({
            'test_type': test['test_type'],
            'plan_name': test['name'],
            'error': (
                f'{test["name"]} could not '
                f'complete: {str(e)}'
            )
        })

# Apply multiple testing correction
all_results, correction_info = (
    apply_correction_if_needed(all_results))
