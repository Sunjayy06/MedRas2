def generate_report(session, df):
    from docx import Document
    from docx.shared import Inches, Pt, RGBColor
    from docx.enum.text import WD_ALIGN_PARAGRAPH
    from docx.oxml.ns import qn
    from docx.oxml import OxmlElement
    import matplotlib.pyplot as plt
    import os, tempfile
    
    doc = Document()
    
    # Set default font
    style = doc.styles['Normal']
    style.font.name = 'Calibri'
    style.font.size = Pt(11)
    
    graph_paths = session.get(
        'graph_paths', [])
    results = session.get('results', [])
    
    # ── SECTION 1: COVER PAGE ──
    doc.add_heading('Statistical Analysis Report', 0)
    p = doc.add_paragraph()
    p.alignment = WD_ALIGN_PARAGRAPH.CENTER
    
    cover_info = [
        ('Study', session.get(
            'objective', 'Not specified')),
        ('Date', session.get(
            'analysis_date', 
            __import__('datetime').date.today()
            .strftime('%d %B %Y'))),
        ('Dataset', session.get(
            'filename', 'Unknown')),
        ('Patients', str(session.get(
            'n_rows', 0))),
        ('Variables', str(session.get(
            'n_cols', 0))),
        ('Generated by', 'MedRAS — Medical '
         'Research Acceleration System')
    ]
    
    for label, value in cover_info:
        p = doc.add_paragraph()
        p.alignment = WD_ALIGN_PARAGRAPH.CENTER
        run = p.add_run(f'{label}: ')
        run.bold = True
        p.add_run(str(value))
    
    doc.add_page_break()
    
    # ── SECTION 2: DATA SUMMARY ──
    doc.add_heading('Data Summary', 1)
    
    variables = session.get('variables', {})
    if variables:
        table = doc.add_table(rows=1, cols=4)
        header = table.rows[0].cells
        for i, h in enumerate([
            'Variable', 'Type', 
            'Missing n', 'Missing %']):
            header[i].text = h
            header[i].paragraphs[0].runs[
                0].bold = True
        
        for col, info in variables.items():
            row = table.add_row().cells
            row[0].text = info.get(
                'display_name', col)
            row[1].text = info.get('type', '')
            missing_n = info.get('missing_n', 0)
            missing_pct = info.get(
                'missing_pct', 0)
            row[2].text = str(missing_n)
            row[3].text = f'{missing_pct:.1f}%'
        
        format_table(table)
        doc.add_paragraph(
            'Table 1a. Variable summary.'
        ).bold = True
    
    cleaning = session.get('cleaning_actions', [])
    if cleaning:
        doc.add_heading('Data Cleaning', 2)
        for action in cleaning:
            doc.add_paragraph(
                action, style='List Bullet')
    
    doc.add_page_break()
    
    # ── SECTION 3: TABLE 1 ──
    doc.add_heading(
        'Table 1. Baseline Characteristics', 1)
    
    table1_data = next(
        (r for r in results 
         if r.get('test_type') == 'descriptive'),
        None)
    
    if table1_data:
        build_table1(doc, table1_data, session)
    
    doc.add_page_break()
    
    # ── SECTION 4: NORMALITY ──
    doc.add_heading('Normality Assessment', 1)
    
    norm_results = session.get(
        'normality_results', {})
    if norm_results:
        table = doc.add_table(rows=1, cols=5)
        for i, h in enumerate([
            'Variable', 'Test', 
            'Statistic', 'p-value', 'Decision']):
            c = table.rows[0].cells[i]
            c.text = h
            c.paragraphs[0].runs[0].bold = True
        
        for var, res in norm_results.items():
            row = table.add_row().cells
            dname = variables.get(var, {}).get(
                'display_name', var)
            row[0].text = dname
            row[1].text = res.get('test', '')
            row[2].text = f"{res.get('stat', ''):.3f}"
            p_val = res.get('p')
            row[3].text = fmt_p(p_val)
            decision = res.get('decision', '')
            row[4].text = decision
            if decision == 'Non-normal':
                for cell in row:
                    for para in cell.paragraphs:
                        for run in para.runs:
                            run.font.color.rgb = (
                                RGBColor(0xA3,0x2D,0x2D))
        
        format_table(table)
        add_caption(doc, 
            'Table 2. Normality assessment results.')
        add_footnote(doc,
            'SW Shapiro-Wilk; KS Kolmogorov-Smirnov. '
            'p < 0.05 indicates non-normal distribution.')
    
    doc.add_page_break()
    
    # ── SECTION 5: PRIMARY ANALYSIS ──
    doc.add_heading('Primary Analysis', 1)
    
    primary = next(
        (r for r in results 
         if r.get('test_type') not in [
             'descriptive', 'pearson', 
             'spearman', 'logistic_regression',
             'linear_regression',
             'kaplan_meier', 'cox_regression',
             'diagnostic_accuracy', 'reliability'
         ] and 'error' not in r),
        None)
    
    if primary:
        build_result_table(
            doc, primary, session, 
            table_num=3)
    
    doc.add_page_break()
    
    # ── SECTION 6: SECONDARY ANALYSES ──
    secondary_types = [
        'pearson', 'spearman',
        'logistic_regression',
        'linear_regression',
        'kaplan_meier', 'cox_regression',
        'diagnostic_accuracy', 'reliability'
    ]
    secondary = [
        r for r in results
        if r.get('test_type') in secondary_types
        and 'error' not in r
    ]
    
    if secondary:
        doc.add_heading('Secondary Analyses', 1)
        table_num = 4
        for res in secondary:
            sub_heading = res.get(
                'plan_name', 
                res.get('test_type', ''))
            doc.add_heading(sub_heading, 2)
            build_result_table(
                doc, res, session, 
                table_num=table_num)
            table_num += 1
            doc.add_paragraph()
        doc.add_page_break()
    
    # ── SECTION 7: FIGURES ──
    if graph_paths:
        doc.add_heading('Figures', 1)
        for i, (path, caption) in enumerate(
                graph_paths, 1):
            if os.path.exists(path):
                doc.add_picture(
                    path, width=Inches(5.5))
                p = doc.add_paragraph(
                    f'Figure {i}. {caption}')
                p.alignment = (
                    WD_ALIGN_PARAGRAPH.CENTER)
                p.runs[0].italic = True
                doc.add_paragraph()
        doc.add_page_break()
    
    # ── SECTION 8: RESULTS SECTION ──
    doc.add_heading('Results Section', 1)
    doc.add_paragraph(
        'Copy the text below directly into '
        'your paper\'s Results section.'
    ).italic = True
    
    results_text = build_results_narrative(
        session, results)
    doc.add_paragraph(results_text)
    
    doc.add_page_break()
    
    # ── SECTION 9: METHODS SECTION ──
    doc.add_heading('Statistical Analysis', 1)
    doc.add_paragraph(
        'Copy the text below directly into '
        'your paper\'s Methods section.'
    ).italic = True
    
    methods_text = build_methods_paragraph(
        session, results)
    doc.add_paragraph(methods_text)
    
    doc.add_page_break()
    
    # ── SECTION 10: LIMITATIONS ──
    doc.add_heading('Limitations and Notes', 1)
    
    limitations = collect_limitations(
        session, results)
    for lim in limitations:
        doc.add_paragraph(
            lim, style='List Bullet')
    
    if not limitations:
        doc.add_paragraph(
            'No major assumption violations '
            'or warnings detected.')
    
    return doc

def format_table(table):
    from docx.oxml.ns import qn
    from docx.oxml import OxmlElement
    from docx.shared import RGBColor, Pt
    
    # Remove ALL vertical borders
    tbl = table._tbl
    tblPr = tbl.get_or_add_tblPr()
    tblBorders = OxmlElement('w:tblBorders')
    for edge in ['left','right','insideV']:
        tag = OxmlElement(f'w:{edge}')
        tag.set(qn('w:val'), 'nil')
        tblBorders.append(tag)
    tblPr.append(tblBorders)
    
    # Horizontal borders only
    for edge in ['top','bottom','insideH']:
        tag = OxmlElement(f'w:{edge}')
        tag.set(qn('w:val'), 'single')
        tag.set(qn('w:sz'), '6')
        tag.set(qn('w:color'), '000000')
        tblBorders.append(tag)
    
    # Remove vertical from every cell
    for i, row in enumerate(table.rows):
        for cell in row.cells:
            tc = cell._tc
            tcPr = tc.get_or_add_tcPr()
            tcBorders = OxmlElement('w:tcBorders')
            for side in ['left','right','insideV']:
                b = OxmlElement(f'w:{side}')
                b.set(qn('w:val'), 'nil')
                tcBorders.append(b)
            tcPr.append(tcBorders)
            
            # Alternating row shading
            if i > 0 and i % 2 == 0:
                shd = OxmlElement('w:shd')
                shd.set(qn('w:fill'), 'F2F2F2')
                shd.set(qn('w:val'), 'clear')
                tcPr.append(shd)
        
        # Bold header row
        if i == 0:
            for cell in row.cells:
                for para in cell.paragraphs:
                    for run in para.runs:
                        run.bold = True
    
    # Set font size for all cells
    from docx.shared import Pt
    for row in table.rows:
        for cell in row.cells:
            for para in cell.paragraphs:
                for run in para.runs:
                    run.font.size = Pt(10)


def add_caption(doc, text):
    from docx.shared import Pt
    p = doc.add_paragraph(text)
    p.runs[0].bold = True
    p.runs[0].font.size = Pt(10)


def add_footnote(doc, text):
    from docx.shared import Pt
    p = doc.add_paragraph(text)
    p.runs[0].italic = True
    p.runs[0].font.size = Pt(9)


def fmt_p(p):
    if p is None: return '-'
    if p != p: return '-'
    if p < 0.001: return '< 0.001'
    return f'{p:.3f}'


def build_table1(doc, table1_data, session):
    variables = session.get('variables', {})
    grouping = session.get('grouping_variable')
    
    if grouping:
        groups = table1_data.get('groups', [])
        headers = ['Variable', 'Overall'] + [
            str(g) for g in groups] + ['p-value']
    else:
        headers = ['Variable', 
                   'Value (n=all)']
    
    table = doc.add_table(rows=1, 
                          cols=len(headers))
    header_row = table.rows[0].cells
    for i, h in enumerate(headers):
        header_row[i].text = h
    
    rows_data = table1_data.get('rows', [])
    for item in rows_data:
        row = table.add_row().cells
        dname = variables.get(
            item.get('variable',''), {}
        ).get('display_name', 
               item.get('variable',''))
        row[0].text = dname
        row[1].text = item.get('overall', '')
        if grouping:
            for j, g in enumerate(groups):
                row[j+2].text = item.get(
                    f'group_{g}', '')
            p_val = item.get('p')
            p_cell = row[-1]
            p_cell.text = fmt_p(p_val)
            if p_val and p_val < 0.05:
                for run in (p_cell.paragraphs[0]
                            .runs):
                    run.bold = True
    
    format_table(table)
    add_caption(doc, 
        'Table 1. Baseline characteristics.')
    add_footnote(doc,
        'Values expressed as mean ± SD, '
        'median (IQR), or n (%) as appropriate. '
        'Independent t-test, Mann-Whitney U, '
        'chi-square, or Fisher exact used '
        'for group comparisons as appropriate. '
        'p < 0.05 considered significant.')


def build_result_table(doc, result, 
                        session, table_num):
    test_name = result.get(
        'plan_name', 
        result.get('test', ''))
    
    # Warning box if present
    warning = result.get('warning')
    if warning:
        p = doc.add_paragraph(
            f'⚠ {warning}')
        p.runs[0].font.color.rgb = (
            __import__('docx.shared', 
                       fromlist=['RGBColor']
            ).RGBColor(0x85,0x4F,0x0B))
    
    # Build table based on result type
    test_type = result.get('test_type', '')
    
    if test_type in [
            'independent_ttest', 'welch_ttest',
            'mann_whitney', 'paired_ttest',
            'wilcoxon', 'anova', 'kruskal_wallis',
            'rm_anova', 'friedman']:
        
        headers = ['Test', 'Statistic', 
                   'df', 'p-value', 
                   'Effect size']
        table = doc.add_table(
            rows=2, cols=len(headers))
        for i, h in enumerate(headers):
            table.rows[0].cells[i].text = h
        row = table.rows[1].cells
        row[0].text = test_name
        stat = result.get('statistic', '')
        row[1].text = (f'{stat:.3f}' 
                       if stat else '-')
        row[2].text = str(result.get('df', '-'))
        p_val = result.get('p')
        row[3].text = fmt_p(p_val)
        if p_val and p_val < 0.05:
            for run in row[3].paragraphs[0].runs:
                run.bold = True
        eff = result.get('effect', '')
        eff_name = result.get('effect_name', '')
        row[4].text = (
            f'{eff_name} = {eff:.3f}' 
            if eff else '-')
        format_table(table)
    
    elif test_type in [
            'logistic_regression',
            'cox_regression']:
        
        headers = ['Variable', 'OR/HR',
                   '95% CI', 'p-value']
        table = doc.add_table(
            rows=1, cols=len(headers))
        for i, h in enumerate(headers):
            table.rows[0].cells[i].text = h
        
        rows_data = result.get('rows', [])
        for item in rows_data:
            row = table.add_row().cells
            row[0].text = item.get('variable','')
            metric = item.get(
                'OR', item.get('HR', '-'))
            row[1].text = (f'{metric:.3f}' 
                          if metric else '-')
            ci_lo = item.get('CI_lo', '')
            ci_hi = item.get('CI_hi', '')
            row[2].text = (
                f'{ci_lo:.3f} – {ci_hi:.3f}'
                if ci_lo and ci_hi else '-')
            p_val = item.get('p')
            row[3].text = fmt_p(p_val)
            if p_val and p_val < 0.05:
                for run in (row[3].paragraphs[0]
                            .runs):
                    run.bold = True
        
        format_table(table)
    
    elif test_type == 'diagnostic_accuracy':
        
        TP = result.get('TP', 0)
        TN = result.get('TN', 0)
        FP = result.get('FP', 0)
        FN = result.get('FN', 0)
        n = TP + TN + FP + FN
        
        headers = ['Measure', 'Value', '95% CI']
        table = doc.add_table(
            rows=1, cols=3)
        for i, h in enumerate(headers):
            table.rows[0].cells[i].text = h
        
        auc = result.get('auc', 0)
        
        from statsmodels.stats.proportion import (
            proportion_confint)
        def wci(k, n):
            lo, hi = proportion_confint(
                int(k), int(n), 
                method='wilson')
            return f'{lo:.3f} – {hi:.3f}'
        
        measures = [
            ('Sensitivity', 
             TP/(TP+FN) if (TP+FN)>0 else 0,
             wci(TP, TP+FN)),
            ('Specificity',
             TN/(TN+FP) if (TN+FP)>0 else 0,
             wci(TN, TN+FP)),
            ('PPV',
             TP/(TP+FP) if (TP+FP)>0 else 0,
             wci(TP, TP+FP)),
            ('NPV',
             TN/(TN+FN) if (TN+FN)>0 else 0,
             wci(TN, TN+FN)),
            ('Accuracy',
             (TP+TN)/n if n>0 else 0,
             wci(TP+TN, n)),
            ('AUC',
             auc, '-')
        ]
        
        for name, val, ci in measures:
            row = table.add_row().cells
            row[0].text = name
            row[1].text = f'{val:.3f}'
            row[2].text = ci
        
        format_table(table)
    
    add_caption(doc,
        f'Table {table_num}. '
        f'{test_name} results.')
    add_footnote(doc,
        'CI confidence interval; '
        'OR odds ratio; HR hazard ratio; '
        'AUC area under curve. '
        'Bold p-values indicate p < 0.05.')


def build_results_narrative(session, results):
    import datetime
    
    n = session.get('n_rows', 0)
    outcome = session.get('outcome_variable','')
    variables = session.get('variables', {})
    grouping = session.get('grouping_variable','')
    
    out_name = variables.get(
        outcome, {}
    ).get('display_name', outcome)
    grp_name = variables.get(
        grouping, {}
    ).get('display_name', grouping)
    
    text = f'A total of {n} participants '
    text += 'were included in the analysis. '
    
    # Normality summary
    norm = session.get('normality_results', {})
    non_normal = [
        variables.get(v,{}).get('display_name',v)
        for v, r in norm.items()
        if not r.get('normal', True)
    ]
    if non_normal:
        text += (
            f'{", ".join(non_normal)} '
            f'{"was" if len(non_normal)==1 else "were"} '
            f'not normally distributed and '
            f'non-parametric tests were used. ')
    
    # Primary result
    primary = next(
        (r for r in results
         if r.get('test_type') not in [
             'descriptive','pearson','spearman',
             'logistic_regression',
             'linear_regression',
             'kaplan_meier','cox_regression',
             'diagnostic_accuracy','reliability'
         ] and 'error' not in r),
        None)
    
    if primary:
        test_name = primary.get('plan_name','')
        stat = primary.get('statistic')
        p = primary.get('p')
        eff = primary.get('effect')
        eff_name = primary.get('effect_name','')
        sig = 'significantly' if (
            p and p < 0.05) else 'not significantly'
        text += (
            f'{out_name} was {sig} different '
            f'across {grp_name} groups '
            f'({test_name}: ')
        if stat:
            text += f'statistic = {stat:.3f}, '
        text += f'p = {fmt_p(p)}'
        if eff:
            text += f', {eff_name} = {eff:.3f}'
        text += ') (Table 3). '
    
    # Regression result
    reg = next(
        (r for r in results
         if r.get('test_type') in [
             'logistic_regression',
             'linear_regression']
         and 'error' not in r),
        None)
    if reg:
        rows = reg.get('rows', [])
        sig_vars = [
            r for r in rows
            if r.get('p') and r['p'] < 0.05
        ]
        if sig_vars:
            v = sig_vars[0]
            metric_key = 'OR' if 'OR' in v else 'HR'
            metric = v.get(metric_key, '')
            text += (
                f'On multivariate analysis, '
                f'{v.get("variable","")} was an '
                f'independent predictor of '
                f'{out_name} '
                f'(adjusted {metric_key} = '
                f'{metric:.3f}, 95% CI: '
                f'{v.get("CI_lo",""):.3f}–'
                f'{v.get("CI_hi",""):.3f}, '
                f'p = {fmt_p(v.get("p"))}). ')
    
    # Survival result
    surv = next(
        (r for r in results
         if r.get('test_type') == 'kaplan_meier'
         and 'error' not in r),
        None)
    if surv:
        p_lr = surv.get('logrank_p')
        sig = 'significant' if (
            p_lr and p_lr < 0.05
        ) else 'no significant'
        text += (
            f'Log-rank test showed {sig} '
            f'difference in survival '
            f'(p = {fmt_p(p_lr)}). ')
    
    return text


def build_methods_paragraph(session, results):
    import scipy, statsmodels, pingouin
    import lifelines
    
    sp_v = scipy.__version__
    sm_v = statsmodels.__version__
    pg_v = pingouin.__version__
    ll_v = lifelines.__version__
    
    norm = session.get('normality_results', {})
    tests_used = list(set([
        r.get('plan_name','')
        for r in results
        if r.get('plan_name')
    ]))
    
    normal_vars = [
        v for v,r in norm.items()
        if r.get('normal', True)
    ]
    non_normal_vars = [
        v for v,r in norm.items()
        if not r.get('normal', True)
    ]
    
    norm_test = 'Shapiro-Wilk' if session.get(
        'n_rows', 0) < 50 else (
        'Kolmogorov-Smirnov')
    
    correction = session.get(
        'correction_info')
    
    text = (
        f'All statistical analyses were performed '
        f'using Python with the scipy '
        f'(v{sp_v}), statsmodels (v{sm_v}), '
        f'pingouin (v{pg_v}), and lifelines '
        f'(v{ll_v}) libraries. '
        f'A p-value < 0.05 was considered '
        f'statistically significant. '
        f'All tests were two-tailed. ')
    
    if normal_vars:
        text += (
            f'Continuous variables were assessed '
            f'for normality using the '
            f'{norm_test} test. '
            f'Normally distributed variables '
            f'were expressed as mean ± standard '
            f'deviation. ')
    
    if non_normal_vars:
        text += (
            f'Non-normally distributed variables '
            f'were expressed as median with '
            f'interquartile range. ')
    
    text += (
        f'Categorical variables were expressed '
        f'as frequencies and percentages. ')
    
    if tests_used:
        text += (
            f'Statistical tests used included: '
            f'{", ".join(tests_used)}. ')
    
    missing = session.get('missing_handled', [])
    if missing:
        text += (
            f'Missing data in '
            f'{", ".join(missing)} were handled '
            f'using '
            f'{session.get("missing_method","mean imputation")}. ')
    
    if correction:
        text += (
            f'Correction for multiple comparisons '
            f'was applied using the '
            f'{correction.get("method","")} method '
            f'({correction.get("n_tests","")} tests). '
            f'Both uncorrected and corrected '
            f'p-values are reported. ')
    
    return text


def collect_limitations(session, results):
    limitations = []
    
    # Power warning
    n = session.get('n_rows', 0)
    if n < 30:
        limitations.append(
            f'Small sample size (n={n}). '
            f'Results should be interpreted '
            f'with caution and confirmed in '
            f'larger studies.')
    
    # Assumption violations
    for r in results:
        if r.get('ph_note'):
            limitations.append(
                f'Cox regression: '
                f'{r.get("ph_note")}')
        if r.get('high_vif_warning'):
            for var, vif in r[
                    'high_vif_warning'].items():
                limitations.append(
                    f'Multicollinearity: {var} '
                    f'(VIF={vif:.1f}) was removed '
                    f'from the regression model.')
        if r.get('heteroscedasticity_warning'):
            limitations.append(
                'Heteroscedasticity detected in '
                'linear regression residuals.')
        if r.get('warning'):
            limitations.append(r['warning'])
        if r.get('error'):
            limitations.append(
                f'{r.get("plan_name","Test")} '
                f'could not complete: '
                f'{r.get("error")}')
    
    # Missing data
    missing_pct = max([
        v.get('missing_pct', 0)
        for v in session.get(
            'variables', {}).values()
    ] or [0])
    if missing_pct > 20:
        limitations.append(
            f'High missing data rate detected '
            f'(up to {missing_pct:.0f}%). '
            f'Results may be affected by '
            f'missing data.')
    
    # Multiple testing
    correction = session.get('correction_info')
    if not correction:
        n_tests = len([
            r for r in results
            if 'error' not in r
        ])
        if n_tests >= 3:
            limitations.append(
                f'{n_tests} statistical tests '
                f'were performed. Multiple '
                f'comparisons increase the risk '
                f'of type I error.')
    
    return limitations
