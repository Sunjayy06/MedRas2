Wilson method:

from statsmodels.stats.proportion import (
    proportion_confint)

def wilson_ci(count, nobs):
    if nobs == 0:
        return (0.0, 1.0)
    lo, hi = proportion_confint(
        count=int(count),
        nobs=int(nobs),
        alpha=0.05,
        method='wilson'
    )
    return (round(lo, 4), round(hi, 4))

Replace every existing proportion_confint

from docx.oxml.ns import qn
from docx.oxml import OxmlElement

def set_cell_border(cell, **kwargs):
    tc = cell._tc
    tcPr = tc.get_or_add_tcPr()
    tcBorders = OxmlElement('w:tcBorders')
    for edge in ('left', 'right', 
                 'insideH', 'insideV'):
        tag = OxmlElement(f'w:{edge}')
        tag.set(qn('w:val'), 'nil')
        tag.set(qn('w:sz'), '0')
        tag.set(qn('w:space'), '0')
        tag.set(qn('w:color'), 'auto')
        tcBorders.append(tag)
    tcPr.append(tcBorders)

def set_table_borders(table):
    # Top border on first row
    # Bottom border on last row  
    # Below-header border on row index 1
    # No vertical borders anywhere
    
    tbl = table._tbl
    tblPr = tbl.get_or_add_tblPr()
    tblBorders = OxmlElement('w:tblBorders')
    
    for edge in ['left', 'right', 
                 'insideV']:
        tag = OxmlElement(f'w:{edge}')
        tag.set(qn('w:val'), 'nil')
        tblBorders.append(tag)
    
    for edge in ['top', 'bottom']:
        tag = OxmlElement(f'w:{edge}')
        tag.set(qn('w:val'), 'single')
        tag.set(qn('w:sz'), '8')
        tag.set(qn('w:color'), '000000')
        tblBorders.append(tag)
    
    tblPr.append(tblBorders)
    
    # Remove vertical borders from 
    # every cell individually
    for row in table.rows:
        for cell in row.cells:
            set_cell_border(cell)
    
    # Add border below header row
    header_row = table.rows[0]
    for cell in header_row.cells:
        tc = cell._tc
        tcPr = tc.get_or_add_tcPr()
        tcBorders = OxmlElement('w:tcBorders')
        bottom = OxmlElement('w:bottom')
        bottom.set(qn('w:val'), 'single')
        bottom.set(qn('w:sz'), '8')
        bottom.set(qn('w:color'), '000000')
        tcBorders.append(bottom)
        tcPr.append(tcBorders)

FIX R9 — Chatbox JSON parsing with 
regex safety net
════════════════════════════════════

In the frontend JavaScript for 
chatbox 3 (Plan and Run screen):

Replace the current JSON.parse call 
with this safer version:

function parseAIAction(responseText) {
    // First try direct parse
    try {
        return JSON.parse(responseText);
    } catch (e) {
        // Try extracting JSON from text
        // AI sometimes wraps in markdown
        const jsonMatch = responseText.match(
            /\{[\s\S]*"action"[\s\S]*\}/
        );
        if (jsonMatch) {
            try {
                return JSON.parse(jsonMatch[0]);
            } catch (e2) {
                return null;
            }
        }
        return null;
    }
}

function handleAIResponse(responseText) {
    const action = parseAIAction(responseText);
    
    if (action && action.action) {
        // Valid action found
        if (action.action === 'add_test') {
            addTestToPlan(action.test_id, 
                         action.reason);
            showChatMessage(
                `Added: ${action.test_id}. 
                ${action.reason}`, 
                'action');
        } else if (action.action === 
                   'remove_test') {
            removeTestFromPlan(action.test_id);
            showChatMessage(
                `Removed: ${action.test_id}.`,
                'action');
        }
    } else {
        // No JSON action found
        // Just show the text response as is
        // Extract text before any JSON block
        const textPart = responseText
            .replace(/```[\s\S]*?```/g, '')
            .replace(/\{[\s\S]*\}/g, '')
            .trim();
        
        if (textPart.length > 0) {
            showChatMessage(textPart, 'ai');
        } else {
            // Fallback if response is 
            // only JSON with no text
            showChatMessage(
                "I understood your request. " +
                "If you want to add or remove " +
                "a test, use the tick buttons " +
                "on the test cards above.",
                'ai');
        }
    }
}

Replace all chatbox 3 AI response 
handling with handleAIResponse().
