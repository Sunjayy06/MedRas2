import os, json, re
import urllib.request

def call_gemini(screen, message, 
                session, api_key):
    
    system_prompts = {
        'normality': (
            'You are a normality explanation '
            'assistant for a medical research '
            'platform. Explain in plain English '
            'only. Never produce any number or '
            'calculation. Never change the '
            'normality decision made by the '
            'engine. If asked to calculate say: '
            'The statistical engine handles all '
            'calculations. I can only explain.'
        ),
        'plan': (
            'You are an analysis planning '
            'assistant. Help researchers '
            'understand and modify their plan. '
            'When user asks to ADD a test '
            'respond with exactly this JSON: '
            '{"action":"add_test",'
            '"test_id":"[name]",'
            '"message":"Added [name]."} '
            'When REMOVING a test: '
            '{"action":"remove_test",'
            '"test_id":"[name]",'
            '"message":"Removed [name]."} '
            'For all other questions respond '
            'with plain text explanation only. '
            'Never produce any p-value, OR, HR, '
            'or any statistical number.'
        ),
        'results': (
            'You are a results interpretation '
            'assistant. The results were '
            'calculated by validated Python '
            'libraries and are correct. '
            'Never recalculate. Never dispute '
            'any result. Never produce a new '
            'number. If asked to recalculate '
            'say exactly: These results were '
            'produced by validated statistical '
            'libraries and cannot be changed. '
            'I can help you understand them. '
            'Explain in plain English suitable '
            'for a clinician with no statistics '
            'background.'
        )
    }
    
    system = system_prompts.get(
        screen, system_prompts['results'])
    context = build_context(screen, session)
    
    full_prompt = (
        system + '\n\n' +
        context + '\n\n' +
        'User question: ' + message
    )
    
    url = (
        'https://generativelanguage.googleapis'
        '.com/v1beta/models/'
        'gemini-1.5-flash:generateContent'
        '?key=' + api_key
    )
    
    payload = json.dumps({
        'contents': [{
            'parts': [{'text': full_prompt}]
        }],
        'generationConfig': {
            'maxOutputTokens': 400,
            'temperature': 0.3
        }
    }).encode('utf-8')
    
    req = urllib.request.Request(
        url,
        data=payload,
        headers={
            'Content-Type': 'application/json'
        },
        method='POST'
    )
    
    try:
        with urllib.request.urlopen(
                req, timeout=10) as resp:
            result = json.loads(
                resp.read().decode())
            text = (
                result['candidates'][0]
                ['content']['parts'][0]['text']
            )
            
            if screen == 'plan':
                match = re.search(
                    r'\{.*?\}', text, re.DOTALL)
                if match:
                    try:
                        action = json.loads(
                            match.group())
                        if 'action' in action:
                            return action
                    except Exception:
                        pass
            
            return {'message': text}
    
    except Exception:
        return rule_based_response(
            screen, message, session)


def build_context(screen, session):
    if screen == 'normality':
        norm = session.get(
            'normality_results', {})
        normal = [
            v for v, r in norm.items()
            if r.get('normal', True)]
        non_normal = [
            v for v, r in norm.items()
            if not r.get('normal', True)]
        return (
            f'Normality context: '
            f'{len(normal)} variables normal, '
            f'{len(non_normal)} non-normal. '
            f'Non-normal: '
            f'{", ".join(non_normal) or "none"}.'
        )
    
    elif screen == 'plan':
        tests = session.get(
            'planned_tests', [])
        names = [
            t.get('label', '') 
            for t in tests]
        return (
            f'Plan context: '
            f'{len(tests)} tests planned: '
            f'{", ".join(names)}.'
        )
    
    elif screen == 'results':
        results = session.get('results', [])
        primary = next(
            (r for r in results
             if 'error' not in r), None)
        if primary:
            p = primary.get('p')
            name = primary.get(
                'plan_name', 'Primary test')
            p_str = (
                '< 0.001' if p and p < 0.001
                else f'{p:.3f}' if p
                else 'N/A')
            return (
                f'Results context: '
                f'Primary result is {name}, '
                f'p = {p_str}.')
        return 'Results context: Analysis complete.'
    
    return ''


def rule_based_response(screen, message, session):
    msg = message.lower()
    
    if screen == 'normality':
        if any(w in msg for w in [
                'non-normal', 'not normal',
                'non normal']):
            return {'message': (
                'Non-normal means your data '
                'does not follow a bell curve. '
                'This matters because t-test '
                'and ANOVA assume normality. '
                'When data is non-normal we '
                'use non-parametric tests like '
                'Mann-Whitney or Kruskal-Wallis '
                'which make no assumption about '
                'distribution shape.'
            )}
        elif 'qq' in msg:
            return {'message': (
                'A QQ plot compares your data '
                'to a perfect normal distribution. '
                'If points follow the diagonal '
                'line your data is normal. '
                'Points curving away indicate '
                'non-normality.'
            )}
        elif 'skew' in msg:
            return {'message': (
                'Skewness measures how asymmetric '
                'your data is. Values between -2 '
                'and +2 are acceptable. Positive '
                'skewness means a long tail to '
                'the right. Negative means a '
                'long tail to the left.'
            )}
        elif any(w in msg for w in [
                'transform', 'log']):
            return {'message': (
                'Log transformation reduces '
                'right skewness and brings data '
                'closer to normal distribution. '
                'This allows parametric tests '
                'to be used. It is noted in '
                'your Methods section.'
            )}
        else:
            return {'message': (
                'I can explain normality results, '
                'QQ plots, skewness, kurtosis, '
                'and transformations. What would '
                'you like to understand?'
            )}
    
    elif screen == 'plan':
        if 'add' in msg and 'survival' in msg:
            return {
                'action': 'add_test',
                'test_id': 'kaplan_meier',
                'message': (
                    'Added Kaplan-Meier survival '
                    'analysis to your plan.')
            }
        elif 'add' in msg and 'regression' in msg:
            return {
                'action': 'add_test',
                'test_id': 'logistic_regression',
                'message': (
                    'Added logistic regression '
                    'to your plan.')
            }
        elif 'remove' in msg and 'regression' in msg:
            return {
                'action': 'remove_test',
                'test_id': 'logistic_regression',
                'message': (
                    'Removed regression from '
                    'your plan.')
            }
        elif any(w in msg for w in [
                'kruskal', 'mann', 'whitney']):
            return {'message': (
                'Kruskal-Wallis compares medians '
                'across 3 or more groups without '
                'assuming normality. Mann-Whitney '
                'compares two groups. Both are '
                'non-parametric alternatives '
                'selected because your outcome '
                'is not normally distributed.'
            )}
        elif 't-test' in msg and 'mann' in msg:
            return {'message': (
                't-test assumes normality and '
                'compares means. Mann-Whitney '
                'makes no normality assumption '
                'and compares medians. For normal '
                'data t-test is more powerful. '
                'For non-normal data Mann-Whitney '
                'is more appropriate.'
            )}
        else:
            return {'message': (
                'I can explain any test in your '
                'plan or help you add and remove '
                'tests. Try asking: what does '
                'Kruskal-Wallis do, or say '
                'add survival analysis.'
            )}
    
    elif screen == 'results':
        if any(w in msg for w in [
                'p value', 'p-value', 'p =']):
            return {'message': (
                'A p-value is the probability '
                'of getting your result by chance '
                'if there was truly no difference. '
                'p below 0.05 means less than 5% '
                'chance of a false positive — '
                'we call this statistically '
                'significant. Smaller p-values '
                'mean stronger evidence.'
            )}
        elif any(w in msg for w in [
                'odds ratio', ' or ', 'or=']):
            return {'message': (
                'An Odds Ratio above 1 means '
                'the factor increases the odds '
                'of the outcome. OR below 1 is '
                'protective. OR = 2.0 means '
                'twice the odds. If the 95% CI '
                'does not cross 1.0 the result '
                'is statistically significant.'
            )}
        elif any(w in msg for w in [
                'hazard', ' hr ', 'hr=']):
            return {'message': (
                'Hazard Ratio above 1 means '
                'higher risk of the event at '
                'any given time. HR below 1 is '
                'protective. HR = 1.5 means '
                '50% higher hazard compared '
                'to the reference group.'
            )}
        elif any(w in msg for w in [
                'write', 'paper', 'sentence',
                'results section']):
            results = session.get('results', [])
            primary = next(
                (r for r in results
                 if 'error' not in r), None)
            if primary:
                p = primary.get('p')
                name = primary.get(
                    'plan_name', 'The test')
                sig = (
                    'statistically significant'
                    if p and p < 0.05
                    else 'not statistically '
                         'significant')
                p_str = (
                    '< 0.001' if p and p < 0.001
                    else f'{p:.3f}' if p
                    else 'N/A')
                return {'message': (
                    f'For your Results section: '
                    f'"{name} showed a {sig} '
                    f'result (p = {p_str}). '
                    f'[Insert your group values '
                    f'from the table above]." '
                    f'Use the Copy button on the '

            )}
        elif any(w in msg for w in [

            return {'message': (
                'Statistical significance tells '

                'tells you if the difference '

function toggleChatbox(screen) {
  const panel = document.getElementById(
    'chatbox-panel-' + screen);
  const isOpen = panel.style.display !== 'none';
  panel.style.display = isOpen ? 'none' : 'flex';
  panel.style.flexDirection = 'column';
}

function addChatMessage(screen, text, type) {
  const container = document.getElementById(
    'chatbox-messages-' + screen);
  const div = document.createElement('div');
  div.className = 'msg-' + type;
  div.textContent = text;
  container.appendChild(div);
  container.scrollTop = container.scrollHeight;
}

function setChatChips(screen, chips) {
  const container = document.getElementById(
    'chatbox-chips-' + screen);
  container.innerHTML = '';
  chips.forEach(chip => {
    const btn = document.createElement('button');
    btn.className = 'chip';
    btn.textContent = chip;
    btn.onclick = () => {
      document.getElementById(
        'chatbox-input-' + screen
      ).value = chip;
      sendChat(screen);
    };
    container.appendChild(btn);
  });
}

async function sendChat(screen) {
  const input = document.getElementById(
    'chatbox-input-' + screen);
  const message = input.value.trim();
  if (!message) return;
  input.value = '';
  
  addChatMessage(screen, message, 'user');
  
  const forbidden = [
    'calculate', 'compute',
    'give me the p', 'what is the p',
    'run the test', 'recalculate'
  ];
  if (forbidden.some(f => 
      message.toLowerCase().includes(f))) {
    addChatMessage(screen,
      'The statistical engine handles all ' +
      'calculations. I can explain results ' +
      'but cannot calculate or change them.',
      'ai');
    return;
  }
  
  try {
    const res = await fetch(
      '/api/stats/chat/' + screen, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json'
      },
      body: JSON.stringify({
        message: message,
        session_id: window.currentSessionId || ''
      })
    });
    
    const data = await res.json();
    
    if (data.action) {
      handlePlanAction(data);
      addChatMessage(
        screen, data.message, 'action');
    } else {
      addChatMessage(
        screen, data.message, 'ai');
    }
  } catch (e) {
    addChatMessage(screen,
      'Could not connect. Please try again.',
      'ai');
  }
}

function handlePlanAction(data) {
  if (!data.action || !data.test_id) return;
  const card = document.querySelector(
    '[data-test-id="' + data.test_id + '"]');
  if (!card) return;
  if (data.action === 'add_test') {
    card.classList.remove('test-removed');
    const tick = card.querySelector('.test-tick');
    if (tick) tick.classList.add('active');
  } else if (data.action === 'remove_test') {
    card.classList.add('test-removed');
    const tick = card.querySelector('.test-tick');
    if (tick) tick.classList.remove('active');
  }
}

function initChatbox(screen, 
                     openingMessage, chips) {
  addChatMessage(screen, openingMessage, 'ai');
  setChatChips(screen, chips);
}
